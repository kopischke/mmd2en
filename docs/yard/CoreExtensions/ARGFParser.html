def to_files(**options)
  files = []
  until ARGV.empty?   # files passed on command line
    files << File.new(ARGV[0], 'r', **options) rescue nil
    ARGV.shift
  end
  if self.path == '-' # STDIN
    file = self.file.dump(**options) unless self.file.read_blocking?
    file and files << file
  end
  files.compact
end
